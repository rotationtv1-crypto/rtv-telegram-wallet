## 3.2 Payment Flow Architecture

### 3.2.1 Telegram Stars Flow

```
TELEGRAM STARS PAYMENT FLOW

User Intent → Telegram Mini App
      ↓
[1] User taps "Purchase X with Stars"
      ↓
[2] Telegram Bot API: createInvoiceLink()
    {
      title: "RTV Creator Credits",
      description: "1000 AI Credits = 1000 Stars",
      payload: JSON.stringify({
        user_id: rtv_user_id,
        product: "ai_credits_1000",
        amount_rtv: 1000,
        company: "RotationTVAI"
      }),
      currency: "XTR",  // Telegram Stars currency code
      prices: [{ label: "AI Credits", amount: 1000 }]
    }
      ↓
[3] User pays in Telegram native interface
      ↓
[4] Telegram sends pre_checkout_query webhook
    → Cloudflare Worker validates inventory/eligibility
    → Respond OK within 10 seconds (required by Telegram)
      ↓
[5] Telegram sends successful_payment webhook
    → Cloudflare Worker receives:
      {
        invoice_payload: "...",
        telegram_payment_charge_id: "...",
        total_amount: 1000,
        currency: "XTR"
      }
      ↓
[6] Cloudflare Worker → Supabase Edge Function
    → Credit user account: +1000 RTV tokens
    → Log to rtv_token_ledger
    → Stars receipt stored in R2 (audit trail)
      ↓
[7] User receives confirmation in Telegram Bot
    → Balance updated in real-time via Supabase Realtime
```

```javascript
// STARS WEBHOOK HANDLER — Cloudflare Worker
export async function handleStarsPayment(request, env) {
  const update = await request.json();
  
  if (update.message?.successful_payment) {
    const payment = update.message.successful_payment;
    const payload = JSON.parse(payment.invoice_payload);
    
    // Idempotency check via Durable Object
    const dedupKey = payment.telegram_payment_charge_id;
    const alreadyProcessed = await env.WEBHOOK_DEDUP.get(dedupKey);
    if (alreadyProcessed) return new Response('OK');
    
    // Credit RTV tokens
    const { data, error } = await supabase
      .from('rtv_token_ledger')
      .insert({
        user_id: payload.user_id,
        transaction_type: 'earn',
        amount_rtv: payload.amount_rtv,
        amount_stars: payment.total_amount,
        usd_equivalent: payment.total_amount * 0.013, // ~$0.013/star
        company_source: payload.company,
        description: `Stars purchase: ${payload.product}`,
        status: 'confirmed',
        ton_tx_hash: null, // Stars don't have TON hash
        metadata: {
          telegram_charge_id: payment.telegram_payment_charge_id,
          stars_amount: payment.total_amount
        }
      });
    
    // Update user balance
    await supabase.rpc('increment_rtv_balance', {
      p_user_id: payload.user_id,
      p_amount: payload.amount_rtv
    });
    
    // Mark as processed
    await env.WEBHOOK_DEDUP.put(dedupKey, '1', { expirationTtl: 86400 });
    
    // Notify user via Telegram
    await sendTelegramMessage(payload.telegram_id, {
      text: `✅ Payment confirmed!\n\n` +
            `💰 +${payload.amount_rtv} RTV Credits added\n` +
            `⭐ ${payment.total_amount} Stars spent\n` +
            `🏢 Service: ${payload.company}`
    });
    
    return new Response('OK');
  }
}
